% (ax+b)(cx+d)
\newcommand\ExoDoubleDistrib{%
    \SR{a}{1}{10}%
    \SR{b}{1}{10}%
    \SR{c}{1}{10}%
    \SR{d}{1}{10}%
    $(\Ra x \Plus{\Rb})(\Rc x \Plus{\Rd})$%
}
\newcommand\SolDoubleDistrib{%
    \SR{a}{1}{10}%
    \SR{b}{1}{10} %
    \SR{c}{1}{10}%
    \SR{d}{1}{10}%
    \FPeval\Va{trunc((\Ra)*(\Rc),0)}%
    \FPeval\Vb{trunc((\Ra)*(\Rd) + (\Rb)*(\Rc),0)}%
    \FPeval\Vc{trunc((\Rb)*(\Rd),0)}%
    $\Va x^2 \Plus{\Vb} x \Plus{\Vc}$%
}
